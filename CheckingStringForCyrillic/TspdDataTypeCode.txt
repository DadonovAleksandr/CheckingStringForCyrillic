namespace Configurator.Alpha
{
    public enum TspdDataTypeCode
    {
        None = 0,
        SI1, 	// целое размером 1 байт (1)
        UI1, 	// Беззнаковое целое размером 1 байт (2)
        SI2, 	// Знаковое целое размером 2 байта (3)
        UI2, 	// Беззнаковое целое размером 2 байта (4)
        SI4, 	// Знаковое целое размером 4 байта (5)
        UI4, 	// Беззнаковое целое размером 4 байта (6)
        SI8, 	// Знаковое целое размером 8 байт (7)
        UI8, 	// Беззнаковое целое размером 8 байт (8)
        FLT, 	// Значение с плавающей запятой размером 4 байта (9)
        DBL, 	// Значение с плавающей запятой размером 8 байт (10)
        BLN, 	// Логическое значение (11)
        STR8, 	// Текстовая строка в кодировке UTF8 (12)
        STR16	// Текстовая строка в кодировке UTF16 (13)
    }
}
